Sub Stocks()    For Each ws In Worksheets        Dim Ticker As String        Dim Total_Stock_Volume As Variant        Total_Stock_Volume = 0                Dim Open_Price As Double        Open_Price = 0        Dim Close_Price As Double        Close_Price = 0        Dim Yearly_Change As Variant        Yearly_Change = 0        Dim Percent_Change As Double        Percent_Change = 0                Dim Summary_Table_Row As Integer        Summary_Table_Row = 2                ws.Range("J1").Value = "Ticker Symbol"        ws.Range("K1").Value = "Yearly Change"        ws.Range("L1").Value = "Percent Change"        ws.Range("M1").Value = "Total Stock Volume"                Open_Price = ws.Cells(2, 3).Value                For i = 2 To 70926            If ws.Cells(i + 1, 1).Value <> ws.Cells(i, 1).Value Then                Ticker = ws.Cells(i, 1).Value                                Close_Price = ws.Cells(i, 6).Value                Yearly_Change = Close_Price - Open_Price                If Open_Price <> 0 Then                    Percent_Change = (Yearly_Change / Open_Price) * 100                End If                                Total_Stock_Volume = Total_Stock_Volume + ws.Cells(i, 7).Value                                ws.Range("J" & Summary_Table_Row).Value = Ticker                                ws.Range("K" & Summary_Table_Row).Value = Yearly_Change                If (Yearly_Change > 0) Then                    ws.Range("K" & Summary_Table_Row).Interior.ColorIndex = 4                ElseIf (Yearly_Change <= 0) Then                    ws.Range("K" & Summary_Table_Row).Interior.ColorIndex = 3                End If                                ws.Range("L" & Summary_Table_Row).Value = (Str(Percent_Change) & "%")                                ws.Range("M" & Summary_Table_Row).Value = Total_Stock_Volume                                Summary_Table_Row = Summary_Table_Row + 1                                Yearly_Change = 0                Open_Price = ws.Cells(i + 1, 3).Value                            Else                Total_Stock_Volume = Total_Stock_Volume + ws.Cells(i, 7).Value            End If        Next i    Next wsEnd Sub
